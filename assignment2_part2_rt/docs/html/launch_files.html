  1"""
  2Launch file for setting up a Gazebo simulation environment.
  3
  4This script performs the following tasks:
  51. Loads and publishes the robot state using a URDF description.
  62. Publishes joint states for the robot.
  73. Spawns the robot entity in Gazebo at coordinates (2.0, 2.0).
  84. Launches a custom node to control the robot's movement.
  95. Launches Gazebo and RViz2 for visualization and simulation.
 10
 11Usage
 12-----
 13.. code-block:: bash
 14
 15   ros2 launch assignment2_part2_rt gazebo.launch.py
 16
 17Parameters
 18----------
 19model : str, optional
 20    Path to the robot model URDF/XACRO file.
 21    Default: package's urdf/robot4.xacro
 22
 23Dependencies
 24-----------
 25* ROS 2 Foxy or later
 26* Gazebo and RViz2
 27* Package 'assignment2_part2_rt' with necessary URDF, config, and scripts
 28"""
 29
 30import os
 31from launch import LaunchDescription
 32from launch.actions import DeclareLaunchArgument, ExecuteProcess
 33from launch.substitutions import Command, LaunchConfiguration
 34from launch_ros.actions import Node
 35from launch_ros.substitutions import FindPackageShare
 36
 37
 38def generate_launch_description() -> LaunchDescription:
 39    """
 40    Generate the launch description for setting up the simulation.
 41
 42    This function sets up the robot description, publishers, Gazebo environment,
 43    and visualization tools required for the simulation.
 44
 45    Components
 46    ----------
 47    1. Robot state publisher - Broadcasts the robot model
 48    2. Joint state publisher - Publishes joint states
 49    3. Spawn entity - Places the robot in Gazebo
 50    4. Move robot node - Controls robot movement
 51    5. Gazebo - Simulation environment
 52    6. RViz2 - Visualization tool
 53
 54    Returns
 55    -------
 56    LaunchDescription
 57        A complete launch description containing all nodes and processes 
 58        needed for the simulation environment.
 59    """
 60    # Define paths for robot description and RViz configuration
 61    test_robot_description_share: str = FindPackageShare(
 62        package="assignment2_part2_rt"
 63    ).find("assignment2_part2_rt")
 64
 65    default_model_path: str = os.path.join(
 66        test_robot_description_share, "urdf/robot4.xacro"
 67    )
 68    rviz_config_path: str = os.path.join(
 69        test_robot_description_share, "config/rviz.rviz"
 70    )
 71
 72    # Node to publish the robot's state using the URDF description
 73    robot_state_publisher_node = Node(
 74        package="robot_state_publisher",
 75        executable="robot_state_publisher",
 76        parameters=[
 77            {"robot_description": Command(["xacro ", LaunchConfiguration("model")])}
 78        ],
 79    )
 80
 81    # Node to publish joint states for the robot
 82    joint_state_publisher_node = Node(
 83        package="joint_state_publisher",
 84        executable="joint_state_publisher",
 85        name="joint_state_publisher",
 86    )
 87
 88    # Node to spawn the robot entity in Gazebo at position (2.0, 2.0)
 89    spawn_entity = Node(
 90        package="gazebo_ros",
 91        executable="spawn_entity.py",
 92        arguments=[
 93            "-entity",
 94            "my_test_robot",
 95            "-topic",
 96            "/robot_description",
 97            "-x",
 98            "2.0",
 99            "-y",
100            "2.0",
101        ],
102        output="screen",
103    )
104
105    # Node to move the robot using custom commands
106    move_robot_node = Node(
107        package="assignment2_part2_rt",
108        executable="move_robot_node",
109        name="move_robot_node",
110    )
111
112    # Create and return the launch description
113    return LaunchDescription(
114        [
115            # Declare the robot model path argument
116            DeclareLaunchArgument(
117                name="model",
118                default_value=default_model_path,
119                description="Absolute path to robot URDF file",
120            ),
121            # Add nodes to the launch description
122            robot_state_publisher_node,
123            joint_state_publisher_node,
124            spawn_entity,
125            move_robot_node,
126            # Launch Gazebo simulator
127            ExecuteProcess(
128                cmd=[
129                    "gazebo",
130                    "--verbose",
131                    "worlds/empty.world",
132                    "-s",
133                    "libgazebo_ros_factory.so",
134                ],
135                output="screen",
136            ),
137            # Launch RViz2 for visualization
138            ExecuteProcess(cmd=["rviz2", "-d", rviz_config_path], output="screen"),
139        ]
140    )
